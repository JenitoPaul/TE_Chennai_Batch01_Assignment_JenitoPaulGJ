package com.filehandling;

import java.io.File;
import java.io.FileOutputStream;
import java.io.FileWriter;
import java.io.IOException;
import java.util.Scanner;

public class File_Read_Write_Delete {

	public static void main(String[] args) {
		
		File ref = new File("C:\\Users\\Admin\\Desktop\\java\\AVI");
		File ref1 = new File("C:\\Users\\Admin\\Desktop\\java\\AVI\\a.java");
		
		try {
			ref.mkdir();
			ref1.createNewFile();
			
			FileWriter ref2 = new FileWriter(ref1);
			ref2.write("THIS IS MY JAVA PROGRAM");
			ref2.flush();
			
			Scanner ref3  = new Scanner(ref1);
			while(ref3.hasNext()) {
				System.out.println(ref3.nextLine());
				System.exit(0);
			}
				
			ref1.delete();
			
		} catch (Exception e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
		finally {
			System.out.println("Finally");
		}
		
	}

}
